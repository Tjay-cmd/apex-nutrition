rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Product images
    match /products/{allPaths=**} {
      allow read: if true;
      // Allow any authenticated user to write (admin validation is handled client-side)
      allow write: if request.auth != null;
    }

    // Allow authenticated users to upload user profile images
    match /profiles/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // Settings images (logos, favicon)
    match /settings/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
