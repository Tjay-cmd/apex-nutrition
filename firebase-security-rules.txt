rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Profiles collection - user profiles with roles
    match /profiles/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Store settings collection - admin access only
    match /store_settings/{document} {
      allow read, write: if request.auth != null;
    }

    // Settings history collection - admin access only
    match /settings_history/{document} {
      allow read, write: if request.auth != null;
    }

    // Customers collection - admin access only
    match /customers/{customerId} {
      allow read, write: if request.auth != null;

      // Customer activity subcollection
      match /customer_activity/{activityId} {
        allow read, write: if request.auth != null;
      }

      // Customer notes subcollection
      match /customer_notes/{noteId} {
        allow read, write: if request.auth != null;
      }
    }

    // Orders collection - admin access only
    match /orders/{orderId} {
      allow read, write: if request.auth != null;
    }

    // Products collection
    // Public reads (homepage, shop, product pages). Writes require auth.
    match /products/{productId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // Categories collection
    // Public reads for filtering/navigation. Writes require auth.
    match /categories/{categoryId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // Default rule - deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
